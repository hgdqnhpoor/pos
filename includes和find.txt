includes和find
//1）includes
let arr4 = [1, 2, 3, 4, 55];
console.log(arr4.includes(5));//返回的是布尔类型

//2）find返回找到的那一项，不会改变原数组,回调函数中返回true表示找到了，找到后停止循环
let result = arr4.find(function (item, index) {
    //console.log(item);
    console.log(item.toString());
    return item.toString().indexOf(5) > -1;
})
